 SIG_M_D:
 DEC R2
 CJNE R2,#-1D,DISPLAY
 MOV R2,#9D
 DEC R3
 CJNE R3,#-1D,DISPLAY
 MOV R3,#5D
 //VD: 24h ( 2 = H1 ; 4 = H2 )
 SIG_H_D:
 MOV A,R5
 XRL A, #2D 
 JZ DECRE_TH0
 JNZ DECRE_TH1
 DECRE_TH1:
 DEC R4
 CJNE R4,#-1D,DISPLAY
 MOV R4,#9D
 SJMP DECRE_H1_1
 
 DECRE_TH0:
 DEC R4
 CJNE R4,#-1D,DISPLAY
 
 DECRE_H1_1:

 CJNE R5,#-1D,DISPLAY
